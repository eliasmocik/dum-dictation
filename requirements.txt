# dum dictation — pinned runtime + test dependencies (Python 3.12).
# Cross-platform: the engine, audio, hotkey and tray deps install everywhere; the
# OS-native bridges (pyobjc on macOS, pywin32 on Windows) and the MLX homophone LLM
# (Apple Silicon only) are guarded by environment markers, so `pip install -r
# requirements.txt` succeeds on macOS, Windows and Linux alike. `./setup` (or
# setup.ps1 on Windows) installs these into .venv.

# Speech recognition engine (Parakeet via ONNX) — cross-platform
sherpa-onnx==1.13.2

# On-device LLM for the homophone/correction stage — MLX is Apple-Silicon only, so it's
# macOS-only. The stage is lazy-imported + gated behind --llm, so Windows/Linux simply run
# without it (the phonetic + alias layers, the main value, work everywhere).
mlx-lm==0.31.3 ; sys_platform == "darwin"

# Audio capture + WAV I/O — cross-platform (PortAudio)
sounddevice==0.5.5
soundfile==0.14.0
numpy==2.4.6

# Global hotkey + synthetic keystrokes (double-tap hotkey; overlay typing) — cross-platform
pynput==1.8.2

# Menu-bar / system-tray icon for the robust daily-driver launch (cross-platform:
# macOS menu bar, Windows tray, Linux indicator).
pystray==0.19.5
pillow==12.1.0

# macOS native APIs: Quartz (CGEvent keystrokes), AppKit/Cocoa (NSPasteboard, AX)
pyobjc-core==12.2 ; sys_platform == "darwin"
pyobjc-framework-Cocoa==12.2 ; sys_platform == "darwin"
pyobjc-framework-Quartz==12.2 ; sys_platform == "darwin"

# Windows native APIs: win32clipboard (clipboard save/restore), win32gui/win32process
# (focused-app detection). Synthetic Unicode typing uses ctypes SendInput (no dep).
pywin32==308 ; sys_platform == "win32"

# Correction-pipeline + test/bench support
rapidfuzz==3.14.5
jellyfish==1.2.1
jiwer==4.0.0

# Model download (used by ./setup)
huggingface-hub==1.19.0
